CREATE TABLE IF NOT EXISTS orders(
    order_id INTEGER NOT NULL PRIMARY KEY,
	user_id INTEGER NOT NULL,
    totalcost REAL,
	datatime TIMESTAMP DEFAULT CURRENT_TIMESTAMP
  );


CREATE TABLE IF NOT EXISTS orderproducts(
    orderid INTEGER,
    productid INTEGER,
    FOREIGN KEY(orderid) REFERENCES orders(order_id) ON DELETE CASCADE ON UPDATE CASCADE,
    FOREIGN KEY(productid) REFERENCES product(product_id) ON DELETE CASCADE ON UPDATE CASCADE
);
